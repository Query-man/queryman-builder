By Gradle:
[source, subs="attributes"]
compile group: 'org.queryman', name: 'builder', version: '{version}'

By Maven
[source,xml,subs="verbatim,attributes"]
<dependency>
    <groupId>org.queryman</groupId>
    <artifactId>builder</artifactId>
    <version>{version}</version>
</dependency>
